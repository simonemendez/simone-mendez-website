async function syncFromOutlook(btn) { btn.textContent = "⏳ Syncing..."; btn.disabled = true; try { const res = await fetch("/.netlify/functions/sync-emails"); const data = await res.json(); // Log debug info to console console.log("Sync Response:", data); if (data.debug) { console.log("Debug Info:", data.debug); } if (data.applications && data.applications.length > 0) { // Just use whatever we get from the sync - it's the single source of truth const synced = data.applications.map((a, i) => ({ id: a.id != null ? a.id : i + 1, company: a.company || "Unknown", position: a.position || "—", dateApplied: a.dateApplied || "", status: normalizeStatus(a.status || "applied"), notes: a.notes || "" })); saveToStorage(synced); applications = synced; filteredData = [...applications]; renderTable(); updateStats(); createCharts(); updateSyncStatus(); btn.textContent = `✅ Synced! ${synced.length} applications`; showNotification(`Sync complete — ${synced.length} applications found.`); } else { btn.textContent = "⚠️ No applications found"; showNotification("No job application emails found in your inbox.", "error"); console.warn("No applications returned from sync"); } } catch (err) { btn.textContent = "❌ Sync failed — try again"; showNotification("Sync failed: " + err.message, "error"); console.error("Sync Error:", err); } setTimeout(() => { btn.textContent = "🔄 Sync from Outlook"; btn.disabled = false; }, 4000); }
